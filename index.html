const express = require("express"); const cors = require("cors"); const app = express(); app.use(express.json()); app.use(cors()); let certificates = []; app.post("/issue", (req, res) => { const cert = req.body; certificates.push(cert); res.send("Certificate Issued"); }); app.get("/verify/:id", (req, res) => { const cert = certificates.find(c => c.id === req.params.id); if (cert) res.json(cert); else res.send("Not Found"); }); }); } Certificate Validation
